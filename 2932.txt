#include<iostream>
#include<string>
#include<map>
using namespace std;
int main()
{  
    map<char,string> m;   
    m[' '] = "%20";  
    m['!'] = "%21";  
    m['$'] = "%24";  
    m['%'] = "%25";  
    m['('] = "%28";  
    m[')'] = "%29";  
    m['*'] = "%2a";  
    string s;
    for(;getline(cin,s)&&s!="#";)
    {
    	for(int i=0;i<s.length();i++)
    	{
	    	if(!m[s[i]].empty())
	    	cout<<m[s[i]];
	    	else
	    	cout<<s[i];
	    }
	    cout<<endl;
    }
   
    return 0;
}
