#include<iostream>
using namespace std;
int s[200004];
int main()
{
	int n,t,c;
	int sum=0,index=0;
	cin>>n>>t>>c;
	for(int i=1;i<=n;i++)
	cin>>s[i];
	for(int i=1;i<=n-c+1;)
	{
		bool it=true;
		for(int j=i;j<i+c;j++)
		{
			if(s[j]>t)
			{
				it=false;
				index=j;
				break;
			}
		}
		if(it)
		{
		sum++;i++;
		while(i<=n-c+1)
		{
		if(s[i+c-1]<=t)
		{
			sum++;
			i++;
		}
		else
		break;					
		}
		}
		else
		i=index+1;
	}
	cout<<sum<<endl;
	return 0;
}
