#include<cstdio>
#include<cstring>
using namespace std;
double map[1008][1008];
int n,m,a,b;
void floyd()
{
	for(int k=1;k<=n;k++)
	for(int i=1;i<=n;i++)
	for(int j=1;j<=n;j++)
	{
		if(map[i][j]<map[i][k]*map[k][j])
		map[i][j]=map[i][k]*map[k][j];
	}
}
int main()
{
	while(scanf("%d",&n)!=EOF)
	{ 
	for(int i=1;i<=n;i++)
	for(int j=1;j<=n;j++)
	scanf("%lf",&map[i][j]);

	floyd();
	scanf("%d",&m);
	while(m--)
	{
	scanf("%d%d",&a,&b);
	if(map[a][b]==0)
	printf("What a pity!\n");
	else
	printf("%.3lf\n",map[a][b]);
	}
	} 
	return 0; 
}
