你找到规律没？
位子分别在
对于1： 1 3 5 7 9 
对于2： 2 6 10 14 18
对于3： 4 12 20 28 36
。。。
对于数n 它的第一次出现是在 2^(n-1)  随后就分别出现在  （1，3，5，7，9 ...）x 2^(n-1) 的位置
也就是说奇数乘上2^(n-1)就是n会出现的位置
 
#include<stdio.h>
int main()
{
　　　__int64 n,k,i,t;
　　　    while(scanf("%I64d%I64d",&n,&k)&&n&&k)
　　　    {
　　　        for(i=1;i<=n;i++)
　　　        {
　　　            t=k%2;
　　　            if(t==1)
　　　            {
　　　                printf("%I64d\n",i);
　　　                break;
　　　           }
　　　          else k/=2;
　　　        }
　　　   }
　　　   return 0;
　　　}
