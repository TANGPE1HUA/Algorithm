#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int n;
    while(cin>>n)
    {
        int m=n-1;
        for(int i=3;i<=n/2;i++)
        {
            if(n%i==0)
            {
                m=i-1;
                break;
            }
        }
        cout<<m<<endl;
    }
    return 0;
}
