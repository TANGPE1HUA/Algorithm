学习map的使用
#include<iostream>
#include<map>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
	map<string,int> a;
	string s;
	
	for(int n;cin>>n&&n!=0;)
	{
		a.clear();
		for(int i=1;i<=n;i++)
		{
			cin>>s;
			a[s]++;
			
		}
		map<string,int>::iterator it;
		int max=0;
		string ss;
		for(it=a.begin();it!=a.end();it++)
		{
		  if(it->second>max)
		 {
		 	max=it->second;
		 	ss=it->first;
 			
 		} 	
		}
		cout<<ss<<endl;
	}
	return 0;
}
