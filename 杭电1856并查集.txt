#include<iostream>
#include<algorithm>
using namespace std;
int father[100005];
int root[100005];
void init()
{
	for(int i=1;i<=100003;i++)
	{
	father[i]=i; 
	root[i]=1;
	}
}
int find(int x)
{
	if(x!=father[x])
	father[x]=find(father[x]);
	return father[x];
}
void combine(int a,int b)
{
	int m=find(a);
	int n=find(b);
	if(m!=n)
	{
		if(root[m]>=root[n])
		{
		father[n]=m;
     	root[m]+=root[n];	
		}
		else
		{
		father[m]=n;
     	root[n]+=root[m];
		}	
	}
}
int main()
{
	for(int n;cin>>n;)
	{
		int a,b;
		int max=1;
		init();
		for(int i=1;i<=n;i++)
		{
		cin>>a>>b;
		combine(a,b);
		}
		for(int j=1;j<=n;j++)
		{
			if(root[j]>max)
			max=root[j];
		}
		cout<<max<<endl;
		
		
	}
	return 0;
	
}
